## Install

```bash
npm install --save resume-library
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'resume-library'
import 'resume-library/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
